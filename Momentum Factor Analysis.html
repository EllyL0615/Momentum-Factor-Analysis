def gen_weights():
    w = np.random.rand(2)
    return w / np.sum(w)

def port_ret(w):
    return np.sum(w * ARoR)

def port_vol(w):
    return np.sqrt(np.dot(w.T, np.dot(port_DRoR.cov() * 365, w)))

def gen_portfolios(n):
    for _ in range(n):
        w = gen_weights()
        yield w[0], w[1], port_ret(w), port_vol(w)
